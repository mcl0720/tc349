This is a pure CSS3 implementation for a loading animation. Just add these definitions to your stylesheet and use the "loading" class on whatever element you want the animation to appear. I have also created a helper JavaScript library that makes this implementation quickly reusable throughout a structured website.

JavaScript Helper for CSS3 Loading Animation

                    
var LoadingModal = function (jqWrapper, options) {
    this.jqWrapper = jqWrapper;
    if (typeof options !== 'undefined') {
        this.styling = typeof options.styling !== 'undefined' ? options.styling : null;
        this.content = typeof options.content !== 'undefined' ? options.content : null;
        this.background = typeof options.background !== 'undefined' ? options.background : false;
        this.fullscreen = typeof options.fullscreen !== 'undefined' ? options.fullscreen : false;
    }
};

LoadingModal.prototype.startLoading = function () {
    var loadingWrapper = $(document.createElement('div')).addClass('loading-overlay').prop('hidden', true);
    var loadingWrapper = $(document.createElement('div')).addClass('loading-wrapper');
    var loading = $(document.createElement('div')).addClass('loading');
    if (this.styling) {
        loadingWrapper.css(this.styling);
    }
    if (this.background) {
        loadingWrapper.addClass('loading-overlay-background');
        loadingWrapper.addClass('loading-wrapper-background');
    }
    if (this.fullscreen) {
        loadingWrapper.addClass('loading-overlay-fullscreen');
    }
    this.jqWrapper.append(loadingWrapper.append(loadingWrapper.append(loading)));
    loadingWrapper.fadeIn("slow");
};

LoadingModal.prototype.finishLoading = function (callback) {
    var loadingWrapper = this.jqWrapper.find('.loading-overlay');
    loadingWrapper.fadeOut("fast", function () {
        loadingWrapper.remove();
        if (callback) {
            callback();
        }
    });
};
                    
                
                    
.loading-overlay {
    position: absolute;
    top: 0;
    left: 0;
    width: 100%;
    height: 100%;
    z-index: 1000;
    vertical-align: middle;
    text-align: center;
}

    .loading-overlay:before {
        content: '';
        display: inline-block;
        height: 100%;
        vertical-align: middle;
    }

.loading-overlay-background {
    background: rgba(0, 0, 0, .4);
}

.loading-overlay-fullscreen {
    position: fixed;
    z-index: 2000;
}

.loading-wrapper {
    margin: 0px auto;
    position: relative;
    display: inline-block;
    vertical-align: middle;
}

.loading-wrapper-background {
    padding: 15px;
}

Here is a simple JavaScript helper toolkit that makes attaching the loading animation to specific parts of a web page rather simple. There are a few new CSS definitions required for the JavaScript helper to be fully funtional, these are found above. You will also need all the CSS definitions from the pure CSS3 implementation of the loading animation.

Example use: 

                    
var loading = new LoadingModal($("#content"));
loading.startLoading();
loading.finishLoading();
loading = null;     // if you are done with the animation, allow garbage collection to clean up the object

Example with options:

                    
var loading = new LoadingModal($("body"), {fullscreen: true, background: true});

A Simple Notification Similar to a "Toast" on Android

                    
// msg: string
// label: {"success", "default", "primary", "info", "warning", "danger"} (pass in one of the string values)
// duration: int (defualt of 1500ms)
var toast = function (msg, label, duration) {
    duration = typeof duration !== 'undefined' ? duration : 1500;
    $("<div class='label label-" + label + "'><h3 class='toast-white'>" + msg + "</h3></div>")
    .css({
        display: "block",
        opacity: 0.90,
        position: "fixed",
        padding: "7px",
        "text-align": "center",
        width: "270px",
        "word-wrap": "break-word",
        "white-space": "normal",
        "box-shadow": "5px 5px 5px rgba(0,0,0,0.5)",
        left: ($(window).width() - 284) / 2,
        top: $(window).height() / 2,
        "z-index": 1050
    })
    .appendTo($(".container")).delay(duration)
    .fadeOut(400, function () {
        $(this).remove();
    });
}

Made to be used in tandem with Twitter Bootstrap. Provides a simple way to get messages to the user with javascript. Just make sure the "appendTo" is passed an HTML element that makes sense for your purposes.

Example use: 

                    
toast("Put your message here!", "success", 4000);
